  7 /**
  8  * addVirtualTouchEnd
  9  * canceling events when the first pointer moved from starting point.
 10  * @param {*} targetElement
 11  * @param {*} targetFunction
 12  * @param {Boolean} targetCapturing
 13  */
 14 var addVirtualTouchEnd = function(targetElement, targetFunction, targetCapturing) {
 15   var prevTouch = null,
 16     prevTouchX = 0,
 17     prevTouchY = 0;
 18 
 19   targetElement.addEventListener('touchstart', function(targetEvent) {
 20     targetEvent.preventDefault();
 21     if (targetEvent.changedTouches && targetEvent.changedTouches[0]) {
 22       prevTouch = targetEvent.changedTouches[0];
 23       prevTouchX = prevTouch.clientX;
 24       prevTouchY = prevTouch.clientY;
 25     }
 26   }, targetCapturing);
 27 
 28   targetElement.addEventListener('touchend', function(targetEvent) {
 29     targetEvent.preventDefault();
 30 
 31     if (prevTouch === null) {
 32       return;
 33     }
 34 
 35     if (prevTouch && targetEvent.changedTouches && targetEvent.changedTouches[0]) {
 36       var dx = prevTouchX - targetEvent.changedTouches[0].clientX;
 37       var dy = prevTouchY - targetEvent.changedTouches[0].clientY;
 38       if (dx * dx + dy * dy > 400) {
 39         return;
 40       }
 41     }
 42     prevTouch = null;
 43     targetFunction(targetEvent);
 44   }, targetCapturing);
 45 };

 47 /**
 48  * Determine which renderer to use for current running environment
 49  * All Android 4.x device will use css renderer except for Android Chrome
 50  * All iOS devices will run on canvas renderer
 51  * @return {string}
 52  */
 53 var getRenderer = function() {
 54   var userAgent = navigator.userAgent;
 55 
 56   /** Android 2.1 does not work with Canvas, force to use CSS renderer */
 57   if (/Android 2\.1/.test(userAgent)) {
 58     return 'useWebkitCSSRenderer';
 59   }
 60 
 61   /** Android 4.x devices are recommended to run with CSS renderer except for Chrome*/
 62   if (/Android 4/.test(userAgent)) {
 63     if (/Chrome/.test(userAgent)) {
 64       return 'useCanvasRenderer';
 65     } else {
 66       return 'useWebkitCSSRenderer';
 67     }
 68   }
 69 
 70   return 'useCanvasRenderer';
 71 };

 73   /* overriding console methods */
 74   var myTargets = ['log', 'warn', 'error'];
 75   var deviceNativeConsoleFunction = {
 76     log: global.console.log,
 77     warn: global.console.warn,
 78     error: global.console.error

 81   var myFakeFunction = {

 89   var prop;

 96   /**
 97    * Self-defined console function to display logs
 98    * @return {*}
 99    */
100   function fakeConsoleLog() {
101     if (!window['LWFLOADER_ENABLE_DEBUG']) {
102       return;
103     }
104 
105     if (/Android/.test(navigator.userAgent)) {
106       // currently only support one argument
107       if (arguments.length > 1 && /(%o)/.test(arguments[0])) {
108         var newMsg = '';
109         var targetParams = ['privateData', 'data', 'url', 'width', 'height'];
110         for (var i = 0; i < arguments.length; i++) {
111           var str = '';
112           for (var prop in arguments[i]) {
113             if (targetParams.indexOf(prop) !== -1) {
114               str += prop + '=' + arguments[i][prop] + '\n';
115             }
116           }
117           newMsg += str;
118         }
119         arguments[0] = arguments[0].replace(/%o/, newMsg);
120       }
121     }
122 
123     return deviceNativeConsoleFunction.log.apply(global.console, arguments);
124   }
