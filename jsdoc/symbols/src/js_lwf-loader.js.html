159    * @param {object} loaderDataBelongToLwfInstance loader instance data
160    * @return {function} function generates lwf path
161    * @private
162    */
163   LwfLoader.prototype.getLwfMapper_ = function(loaderDataBelongToLwfInstance) {
164     if (loaderDataBelongToLwfInstance['lwfMap'] === null) {
165       return _.bind(this.getLwfPath_, this);
166     }
167 
168     var lwfMap = loaderDataBelongToLwfInstance['lwfMap'];
169     if (_.isObject(lwfMap)) {
170       if (_.isFunction(lwfMap)) {
171         return lwfMap;
172       }

175         var path = lwfId;
176         if (lwfMap.hasOwnProperty(lwfId)) {
177           path = lwfMap[lwfId];
178         }
179         if (!/\.lwf$/.test(path)) {
180           path += '.lwf';
181         }
182         return path;
183       };
184     }
185     throw new Error('unsupported lwfMap');

288     var loaderDataBelongToLwfInstance = privateData['_loaderData'];
289     var loader = loaderDataBelongToLwfInstance['loader'];

293     setting['privateData'] = null;

296       loader.handleLoadError_(this, loaderDataBelongToLwfInstance);
297       return;
298     }
299     lwfInstances.push(lwf);

301     if (_.isFunction(loaderDataBelongToLwfInstance['callback']['onLoad'])) {
302       loaderDataBelongToLwfInstance['callback']['onLoad'](lwf);
303     }
304 
305     var onExec, onMove, onPress, onRelease, onGestureEnd;
306     var stage = this['stage'];
307     var widthInit = 0, heightInit = 0;
308     var stageScale = 1;
309     var stageWidth = 0;
310     var stageHeight = 0;
311     var t0 = global.performance.now();
312 
313     /** for FPS display */
314     var execCount = 0;
315     var t0_60 = t0;
316     var fps_num60 = 0;
317 
318     lwf['rootMovie'].moveTo(rootOffset.x, rootOffset.y);
319 
320     /**
321      * loading handler, set the required information for LWF files
322      */
323     onExec = function() {
324       try {
325         if (!lwf) {
326           return;
327         }
328 
329         var devicePixelRatio = global['devicePixelRatio'];
330         if (lwfRenderer === 'useWebkitCSSRenderer') {
331           devicePixelRatio = 1;
332         }
333 
334         var width = lwf['width'];
335         var height = lwf['height'];
336         var innerWidth = global.innerWidth;
337         var innerHeight = global.innerHeight;
338 
339         if (isAndroid) {
340           /** fix innerWidth/Height for old Android devices */
341           if (global.innerWidth > global.screen.width) {
342             innerWidth = global.screen.width;
343           }
344           if (global.innerHeight > global.screen.height) {
345             innerHeight = global.screen.height;
346           }
347         }
348 
349         if (width > innerWidth || loaderDataBelongToLwfInstance['resizeStretch']) {
350           width = innerWidth;
351         }
352         if (height > innerHeight || loaderDataBelongToLwfInstance['resizeStretch']) {
353           height = innerHeight;
354         }
355 
356         if (widthInit !== width || heightInit !== height) {
357           if (setting['fitForWidth']) {
358             if (setting.widthLimit) {
359               width = (width > setting.widthLimit) ? setting.widthLimit : width;
360             }
361             stageWidth = Math.round(width);
362             stageHeight = Math.round(width * lwf['height'] / lwf['width']);
363           } else {
364             if (setting.heightLimit) {
365               height = (height > setting.heightLimit) ? setting.heightLimit : height;
366             }
367             stageWidth = Math.round(height * lwf['width'] / lwf['height']);
368             stageHeight = Math.round(height);
369           }
370 
371           stageEventReceiver.style.width = stage.style.width = stageWidth + 'px';
372           stageEventReceiver.style.height = stage.style.height = stageHeight + 'px';
373 
374           stageEventReceiver.width = stage.width = ~~(stageWidth * devicePixelRatio);
375           stageEventReceiver.height = stage.height = ~~(stageHeight * devicePixelRatio);
376           if (setting['fitForWidth']) {
377             stageScale = stageWidth / stage.width;
378             lwf.property.clear();
379             lwf.fitForWidth(stage.width, stage.height);
380           } else {
381             stageScale = stageHeight / stage.height;
382             lwf.property.clear();
383             lwf.fitForHeight(stage.width, stage.height);
384           }
385           widthInit = width;
386           heightInit = height;
387 
388           /** set the external div size */
389           if (lwfDivId) {
390             var windowDiv = document.getElementById(lwfDivId);
391             windowDiv.style.width = stageWidth + 'px';
392             windowDiv.style.height = stageHeight + 'px';
393           }
394         }
395 
396         var t1 = global.performance.now();
397         var dt = t1 - t0;
398         t0 = t1;
399         if (!loader.pausing) {
400           lwf.exec(dt / 1000);
401           lwf.render();
402         }
403 
404         global.requestAnimationFrame(onExec);
405         if (execCount % 60 === 0) {
406           fps_num60 = Math.round(60000.0 / (t1 - t0_60));
407           t0_60 = t1;
408           execCount = 0;
409         }
410         if (loader.debug || loaderDataBelongToLwfInstance['debug']) {
411           var divElement = document.getElementById('lwf_info' + loaderDataBelongToLwfInstance['debugInfoElementId']);
412           divElement.innerHTML = fps_num60 + 'fps';
413         }
414         execCount++;
415         loader.currentFPS = fps_num60;
416       } catch (myException) {
417         loader.handleException_(myException, loaderDataBelongToLwfInstance);
418       }
419     };

421     /**
422      * touch move handler
423      * @param {object} myEvent touch event
424      */
425     onMove = function(myEvent) {
426       try {
427         if (isPreventDefaultEnabled) {
428           myEvent.preventDefault();
429         }
430         if (!lwf) {
431           return;
432         }
433 
434         var touchX, touchY;
435         var stageRect = stage.getBoundingClientRect();
436 
437         if (isTouchEventEnabled) {
438           var myTouch = myEvent.touches[0];
439           touchX = myTouch.pageX;
440           touchY = myTouch.pageY;
441         } else {
442           touchX = myEvent.offsetX;
443           touchY = myEvent.offsetY;
444         }
445 
446         if (isSp) {
447           touchX -= stage.offsetLeft;
448           touchY -= stage.offsetTop;
449           touchX = touchX - stageRect.left - window.scrollX;
450           touchY = touchY - stageRect.top - window.scrollY;
451         }
452 
453         touchX /= stageScale;
454         touchY /= stageScale;
455 
456         lwf.inputPoint(touchX, touchY);
457       } catch (myException) {
458         loader.handleException_(myException, loaderDataBelongToLwfInstance);
459       }
460     };

621   /**
622    * load and play LWF to specified DOM element
623    * @param {object} targetElem target DOM element
624    * @param {object} lwfParam sets LWF parameters using optional params
625    * @return {*}
626    */
627   LwfLoader.prototype.playLWF = function(targetElem, lwfParam) {
628     var lwfDisplaySetting = targetElem.getAttribute('data-lwf-display_setting');
629     lwfDisplaySetting = lwfDisplaySetting ? JSON.parse(lwfDisplaySetting) : {};
630     lwfResizeMode = lwfDisplaySetting.resizeMode || lwfResizeMode;
631     lwfResizeStretch = lwfDisplaySetting.resizeStretch || lwfResizeStretch;
632     lwfDivId = lwfDisplaySetting.displayDivId || lwfDivId;
633 
634     /* prepare LWF renderer */
635     var LWF, cache;
636     LWF = global['LWF'];
637     if (!lwfRenderer) {
638       lwfRenderer = lwfDisplaySetting.renderer || global['Utility']['getRenderer']();
639 
640       if (lwfRenderer === 'canvas') {
641         lwfRenderer = 'useCanvasRenderer';
642       } else if (lwfRenderer === 'webkitcss') {
643         lwfRenderer = 'useWebkitCSSRenderer';
644       } else if (lwfRenderer === 'webgl') {
645         lwfRenderer = 'useWebGLRenderer';
646       }
647 
648       if (lwfRenderer.match(/use.+Renderer/i)) {
649         LWF[lwfRenderer]();
650       } else {
651         throw new Error('Renderer parameters are not properly set');
652       }
653     } else if (lwfDisplaySetting.renderer && (lwfRenderer !== lwfDisplaySetting.renderer)) {
654       throw new Error('cannot use multiple renderer! LWF has been initialized for ' + lwfRenderer + ' renderer.');
655     }
656 
657     cache = LWF['ResourceCache']['get']();
658 
659     /** prepare LWF parameters */
660     var myLwfParam = _.isObject(global['lwfDefaultParam']) ? _.clone(global['lwfDefaultParam']) : {};
661 
662     /** whether enables the debugger */
663     myLwfParam['debug'] = myLwfParam['debug'] ? true : false;
664 
665     /** set load event handler */
666     myLwfParam['onload'] = this.onLoad;
667 
668     /** background color setting*/
669     if (!_.isUndefined(lwfDisplaySetting.backgroundColor)) {
670       myLwfParam['setBackgroundColor'] = lwfDisplaySetting.backgroundColor;
671     } else {
672       myLwfParam['useBackgroundColor'] = true;
673     }
674 
675     /** display related setting */
676     myLwfParam['fitForHeight'] = false;
677     myLwfParam['fitForWidth'] = false;
678     myLwfParam[lwfResizeMode] = true;
679     myLwfParam['resizeStretch'] = lwfResizeStretch;
680     myLwfParam['widthLimit'] = lwfDisplaySetting.widthLimit;
681     myLwfParam['heightLimit'] = lwfDisplaySetting.heightLimit;
682 
683     /** web worker setting, only available on Chrome or non-Android devices*/
684     myLwfParam['worker'] = useWebWorker;
685 
686     /** handle buggy css behaviour in certain devices */
687     if (isAndroid && / SC-0/.test(userAgent) && lwfRenderer === 'useWebkitCSSRenderer') {
688       myLwfParam['quirkyClearRect'] = true;
689     }
690 
691     /** force to disable use3D on Android devices */
692     if (isAndroid) {
693       myLwfParam['use3D'] = false;
694     }
695 
696     /** prepare LWF or lwf-loader required data */
697     myLwfParam['privateData'] = {};
698     myLwfParam['pos'] = null;
699     myLwfParam['imageMap'] = {};
700     myLwfParam['soundMap'] = {};
701     myLwfParam['pageTransitionMap'] = {};
702     myLwfParam['buttonEventMap'] = {};
703     myLwfParam['fsCommandMap'] = {};
704     myLwfParam['handler'] = {};
705     myLwfParam['callback'] = {};
706     myLwfParam['lwfMap'] = null;
707 
708     for (var i = 0; i < targetElem.attributes.length; i++) {
709       var elemAttr = targetElem.attributes[i];
710       if (elemAttr.name && elemAttr.name.indexOf('data-lwf-') === 0) {
711         var lwfParamName = elemAttr.name.slice('data-lwf-'.length);
712         lwfParamName = lwfParamName.replace(/_./g, function(matched) {
713           return matched.charAt(1).toUpperCase();
714         });
715 
716         /** 'data-lwf-display_setting' is used by global setting */
717         if (lwfParamName === 'displaySetting') {
718           continue;
719         }
720 
721         var lwfParamValue = elemAttr.value;
722         try {
723           if (lwfParamValue.charAt(0) === '{') {
724             lwfParamValue = JSON.parse(lwfParamValue);
725           }
726         } catch (myException) {
727           /**
728            * exception case handling JSON.parse failure
729            * ignore exception at current version
730            */
731         }
732         myLwfParam[lwfParamName] = lwfParamValue;
733       }
734     }
735 
736     /** for displaying high-resolution images on certain devices */
737     if (global['devicePixelRatio'] == 2 && myLwfParam['useHighResImage']) {
738       myLwfParam['imageSuffix'] = '_@2x';
739     }
740 
741     /** 'data-lwf' is alias of 'data-lwf-lwf' */
742     if (!myLwfParam.hasOwnProperty('lwf')) {
743       var lwfFileName = targetElem.getAttribute('data-lwf');
744       if (_.isString(lwfFileName)) {
745         myLwfParam['lwf'] = lwfFileName;
746       }
747     }
748 
749     /** copy 'name' property to 'lwf' to keep backward compatibility */
750     if (myLwfParam.hasOwnProperty('name')) {
751       myLwfParam['lwf'] = myLwfParam['name'];
752     }
753 
754     if (_.isObject(lwfParam)) {
755       _.extend(myLwfParam, lwfParam);
756     }
757 
758     /** set(extend) LWF parameters by initialize hooks */
759     if (this.initializeHooks.length > 0) {
760       var func = function(v) {
761         if (_.isFunction(v)) {
762           var results = v.call(this, targetElem);
763           for (var key in results) {
764             if (results.hasOwnProperty(key)) {
765               _.extend(myLwfParam[key], results[key]);
766             }
767           }
768         }
769       };
770 
771       _.each(this.initializeHooks, _.bind(func, this));
772     }
773     this.initializeHooks = [];
774 
775     /** check lwf file name */
776     if (!myLwfParam.hasOwnProperty('lwf')) {
777       throw new Error('[LWF] no LWF input');
778     }
779 
780     /** Initialize loader-data parameters*/
781     var loaderDataBelongToLwfInstance = {};
782     loaderDataBelongToLwfInstance['debugInfoElementId'] = ++debugInfoElementId;
783     loaderDataBelongToLwfInstance['setting'] = myLwfParam;
784     loaderDataBelongToLwfInstance['loader'] = this;
785     loaderDataBelongToLwfInstance['debug'] = false;
786     loaderDataBelongToLwfInstance['soundMap'] = null;
787     loaderDataBelongToLwfInstance['pageTransitionMap'] = null;
788     loaderDataBelongToLwfInstance['buttonEventMap'] = null;
789     loaderDataBelongToLwfInstance['fsCommandMap'] = null;
790     loaderDataBelongToLwfInstance['handler'] = null;
791     loaderDataBelongToLwfInstance['callback'] = null;
792     loaderDataBelongToLwfInstance['stageEventReceiver'] = null;
793     loaderDataBelongToLwfInstance['lwfMap'] = null;
794 
795     /** for backward compatibility */
796     var debugFlag = targetElem.getAttribute('data-debug');
797     if (debugFlag) {
798       myLwfParam['privateData']['debug'] = true;
799     }
800 
801     loaderDataBelongToLwfInstance['debug'] = myLwfParam['debug'] ? true : false;
802     delete myLwfParam['debug'];
803 
804     if (!myLwfParam['privateData'].hasOwnProperty('lwfLoader')) {
805       /** for backward compatibility */
806       myLwfParam['privateData']['lwfLoader'] = this; /* pass loader object to LWF */
807     }
808 
809     myLwfParam['imageMap'] = this.getImageMapper_(myLwfParam['imageMap']);
810 
811     if (!_.isEmpty(myLwfParam['soundMap'])) {
812       loaderDataBelongToLwfInstance['soundMap'] = myLwfParam['soundMap'];
813     }
814     delete myLwfParam['soundMap'];
815 
816     if (_.isObject(myLwfParam['pageTransitionMap'])) {
817       loaderDataBelongToLwfInstance['pageTransitionMap'] = myLwfParam['pageTransitionMap'];
818     }
819     delete myLwfParam['pageTransitionMap'];
820 
821     loaderDataBelongToLwfInstance['buttonEventMap'] = myLwfParam['buttonEventMap'];
822     delete myLwfParam['buttonEventMap'];
823 
824     if (!_.isEmpty(myLwfParam['fsCommandMap'])) {
825       loaderDataBelongToLwfInstance['fsCommandMap'] = myLwfParam['fsCommandMap'];
826     }
827     delete myLwfParam['fsCommandMap'];
828 
829     if (!_.isEmpty(myLwfParam['handler'])) {
830       loaderDataBelongToLwfInstance['handler'] = myLwfParam['handler'];
831     }
832     delete myLwfParam['handler'];
833 
834     loaderDataBelongToLwfInstance['callback'] = myLwfParam['callback'];
835     delete myLwfParam['callback'];
836 
837     loaderDataBelongToLwfInstance['lwfMap'] = myLwfParam['lwfMap'];
838     delete myLwfParam['lwfMap'];
839 
840     loaderDataBelongToLwfInstance['resizeStretch'] = myLwfParam['resizeStretch'];
841     delete myLwfParam['resizeStretch'];
842 
843     if (this.isLwfsEnvironment_()) {
844       var lwfPath;
845       if (myLwfParam['prefix']) {
846         lwfPath = myLwfParam['prefix'] + myLwfParam['lwf'];
847       } else {
848         lwfPath = myLwfParam['lwf'];
849       }
850       myLwfParam['prefix'] = lwfPath.slice(0, lwfPath.lastIndexOf('/') + 1);
851       delete myLwfParam['imagePrefix'];
852       myLwfParam['lwf'] = lwfPath.slice(lwfPath.lastIndexOf('/') + 1);
853     }
854 
855     var pos = {};
856 
857     pos['position'] = 'absolute';
858     pos['top'] = 0;
859     pos['left'] = 0;
860 
861     _.extend(pos, myLwfParam['pos']);
862     delete myLwfParam['pos'];
863 
864     /** prepare LWF stage */
865     if (targetElem.style.position === 'static' ||
866       targetElem.style.position === '')
867     {
868       targetElem.style.position = 'relative';
869     }
870     var stage = createStage(lwfRenderer);
871     stage.style.position = pos['position'];
872     stage.style.top = pos['top'] + 'px';
873     stage.style.left = pos['left'] + 'px';
874     stage.style.zIndex = targetElem.style.zIndex + 1;
875     targetElem.appendChild(stage);
876 
877     /** use event receiver for avoiding Galaxy S3's translateZ bug */
878     var stageEventReceiver = null;
879     if (isSp) {
880       stageEventReceiver = document.createElement('div');
881       stageEventReceiver.style.position = 'absolute';
882       stageEventReceiver.style.top = pos['top'] + 'px';
883       stageEventReceiver.style.left = pos['left'] + 'px';
884       stageEventReceiver.style.zIndex = stage.style.zIndex + 1;
885       targetElem.appendChild(stageEventReceiver);
886     } else {
887       stageEventReceiver = stage;
888     }
889     myLwfParam['stage'] = stage;
890     loaderDataBelongToLwfInstance['stageEventReceiver'] = stageEventReceiver;
891 
892     if (this.debug || loaderDataBelongToLwfInstance['debug']) {
893       var divElement = document.createElement('div');
894       divElement.id = 'lwf_info' + debugInfoElementId;
895       divElement.style.position = pos['position'];
896       divElement.style.top = pos['top'] + 'px';
897       divElement.style.left = pos['left'] + 'px';
898       divElement.style.color = 'white';
899       divElement.style.zIndex = 10000;
900       targetElem.appendChild(divElement);
901     }
902 
903     if (!myLwfParam['lwf']) {
904       console.error('[LWF] no LWF input');
905       return;
906     }
907     loaderDataBelongToLwfInstance['useLwfJs'] = /\.lwf.js(\?.*)?$/.test(myLwfParam['lwf']);
908 
909     if (myLwfParam.hasOwnProperty('_loaderData')) {
910       throw new Error('cannot use \'_loaderData\' property that used by lwf-loader!');
911     }
912 
913     myLwfParam['privateData']['_loaderData'] = loaderDataBelongToLwfInstance;
914 
915     cache.loadLWF(myLwfParam);
916   };

918   /**
919    * request LWF to attach
920    * when child LWF is attached to parent, this function will be called
921    * this methods attempts to inherit parent's lwf-loader instance's parameters
922    * @param {object} lwf parent LWF instance
923    * @param {number} lwfId LWF ID
924    * @param {object} imageMap LWF image map
925    * @param {object} privateData LWF private data
926    * @param {function} myCallback callback to return attach LWF instance
927    */
928   LwfLoader.prototype.requestLWF_ = function(lwf, lwfId, imageMap, privateData, myCallback) {
929     var lwfInput = {};
930 
931     /** for LWFS environment, inherit parents' settings from window['testlwf_settings']*/
932     if (this.isLwfsEnvironment_()) {
933       var lwfParam = _.clone(window['testlwf_settings']);
934 
935       var prefix;
936       if (lwfParam['prefix']) {
937         prefix = lwfParam['prefix'] + lwfId + '/_/';
938       } else {
939         prefix = document.location.pathname;
940         prefix = prefix.replace(lwf['name'], lwfId);
941         prefix = prefix.slice(0, prefix.lastIndexOf('/') + 1) + '_/';
942       }
943       lwfParam['prefix'] = prefix;
944       delete lwfParam['imagePrefix'];
945 
946       lwfInput = this.getLwfPath_(lwfId);
947       if (!lwfInput) {
948         console.error('[LWF] no LWF input');
949         myCallback('no LWF input', null);
950         return null;
951       }
952       lwfParam['lwf'] = lwfInput;
953       if (isAndroid) {
954         lwfParam['lwf'] += '.js';
955       }
956       lwfParam['parentLWF'] = lwf;
957       lwfParam['stage'] = lwf['stage'];
958       lwfParam['worker'] = useWebWorker;
959       lwfParam['active'] = false;
960 
961       if (imageMap) {
962         lwfParam['imageMap'] = this.getImageMapper_(imageMap);
963       }
964       if (privateData) {
965         lwfParam['privateData'] = privateData;
966       }
967 
968       lwfParam.onload = function(childLwf) {
969         var loader = this['privateData']['lwfLoader'];
970         if (!childLwf) {
971           loader.handleLoadError_(this);
972           return myCallback(this['error'], childLwf);
973         }
974         return myCallback(null, childLwf);
975       };
976 
977     } else {
978       /** use data and loader parameters stored in parent LWF instance*/
979       var parentPrivateData = lwf['privateData'];
980       if ('undefined' === typeof parentPrivateData) {
981         console.error('[LWF] Parent private data is not found');
982         myCallback('no parents data found', null);
983         return null;
984       }
985 
986       var loaderDataBelongToLwfInstance = _.clone(parentPrivateData['_loaderData']);
987       if ('undefined' === typeof loaderDataBelongToLwfInstance) {
988         console.error('[LWF] Parent Loader data is not found');
989         myCallback('no loader parameters to inherit');
990         return null;
991       }
992 
993       var parentLwfParam = loaderDataBelongToLwfInstance['setting'];
994       lwfParam = _.clone(parentLwfParam); /* inherit from parent LWF */
995 
996       /* child LWFs are rendered using their own size. */
997       lwfParam['fitForHeight'] = lwfParam['fitForWidth'] = false;
998       loaderDataBelongToLwfInstance['setting'] = lwfParam;
999 
1000       lwfInput = null;
1001       try {
1002         lwfInput = (this.getLwfMapper_(loaderDataBelongToLwfInstance))(lwfId);
1003       } catch (myException) {
1004         this.handleException_(myException, loaderDataBelongToLwfInstance);
1005       }
1006 
1007       if (!lwfInput) {
1008         console.error('[LWF] no LWF input');
1009         myCallback('no LWF input', null);
1010         return null;
1011       }
1012       lwfParam['lwf'] = lwfInput;
1013       if (loaderDataBelongToLwfInstance['useLwfJs']) {
1014         lwfParam['lwf'] += '.js';
1015       }
1016 
1017       lwfParam['parentLWF'] = lwf;
1018       lwfParam['stage'] = lwf['stage'];
1019       lwfParam['worker'] = useWebWorker;
1020       lwfParam['active'] = false;
1021 
1022       if (imageMap) {
1023         lwfParam['imageMap'] = this.getImageMapper_(imageMap);
1024       }
1025       if (privateData) {
1026         privateData['_loaderData'] = loaderDataBelongToLwfInstance;
1027         lwfParam['privateData'] = privateData;
1028       }
1029 
1030       lwfParam['onload'] = function(childLwf) {
1031         var loader = loaderDataBelongToLwfInstance['loader'];
1032         if (!childLwf) {
1033           loader.handleLoadError_(this, loaderDataBelongToLwfInstance);
1034           return myCallback(this['error'], childLwf);
1035         }
1036         return myCallback(null, childLwf);
1037       };
1038     }
1039 
1040     return lwfParam;
1041   };
